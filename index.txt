        // 퀴즈 문항을 화면에 동적으로 렌더링하는 함수
        function renderQuiz() {
            quizContainer.innerHTML = '';
            answeredQuestions.clear();
            updateSubmitButtonState();

            quizData.forEach((item, index) => {
                const questionElement = document.createElement('div');
                questionElement.className = 'bg-white p-6 rounded-xl shadow-md space-y-4';
                questionElement.innerHTML = `
                    <p class="text-lg font-bold">Q${index + 1}. ${item.question}</p>
                    <div class="flex flex-row justify-center space-x-8">
                        <div class="relative">
                            <button class="quiz-option btn w-20 h-20 md:w-24 md:h-24 rounded-full font-bold transition duration-300 flex items-center justify-center text-3xl md:text-4xl" data-answer="O" data-question-index="${index}">O</button>
                            <span class="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-20 h-20 md:w-24 md:h-24 rounded-full border-4 border-transparent pointer-events-none transition duration-300"></span>
                        </div>
                        <div class="relative">
                            <button class="quiz-option btn w-20 h-20 md:w-24 md:h-24 rounded-full font-bold transition duration-300 flex items-center justify-center text-3xl md:text-4xl" data-answer="X" data-question-index="${index}">X</button>
                            <span class="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 w-20 h-20 md:w-24 md:h-24 rounded-full border-4 border-transparent pointer-events-none transition duration-300"></span>
                        </div>
                    </div>
                    <div id="feedback-${index}" class="text-base mt-2 hidden"></div>
                `;
                quizContainer.appendChild(questionElement);
            });
            
            // 새로 렌더링된 버튼에 이벤트 리스너를 다시 추가
            document.querySelectorAll('.quiz-option').forEach(button => {
                button.addEventListener('click', handleAnswer);
                button.classList.add('bg-gray-200', 'text-gray-700');
            });
        }
        
        // 제출 버튼의 활성화/비활성화 상태를 업데이트하는 함수
        function updateSubmitButtonState() {
            if (answeredQuestions.size === quizData.length) {
                submitButton.classList.remove('opacity-50', 'cursor-not-allowed');
                submitButton.disabled = false;
            } else {
                submitButton.classList.add('opacity-50', 'cursor-not-allowed');
                submitButton.disabled = true;
            }
        }

        // 퀴즈 정답 선택을 처리하는 함수
        function handleAnswer(event) {
            const selectedButton = event.target.closest('button');
            const questionIndex = parseInt(selectedButton.getAttribute('data-question-index'));
            const questionContainer = selectedButton.closest('.bg-white');
            const selectedAnswer = selectedButton.getAttribute('data-answer');
            const correctAnswer = quizData[questionIndex].answer;
            const feedbackElement = document.getElementById(`feedback-${questionIndex}`);
            const buttons = questionContainer.querySelectorAll('.quiz-option');

            // 버튼 스타일 초기화
            buttons.forEach(btn => {
                btn.classList.remove('bg-primary-blue', 'bg-orange', 'text-white');
                btn.classList.add('bg-gray-200', 'text-gray-700');
                btn.nextElementSibling.classList.remove('border-primary-blue', 'border-orange');
            });

            // 선택된 버튼에 스타일 적용 및 피드백 메시지 표시
            if (selectedAnswer === correctAnswer) {
                // 정답인 경우
                selectedButton.classList.remove('bg-gray-200', 'text-gray-700');
                selectedButton.classList.add('bg-primary-blue', 'text-white');
                selectedButton.nextElementSibling.classList.add('border-primary-blue');
                feedbackElement.innerHTML = `<span class="text-primary-blue font-bold">정답입니다!</span><br>${quizData[questionIndex].explanation}`;
                
                // 모든 버튼을 비활성화하고 정답 세트에 추가
                buttons.forEach(btn => {
                    btn.disabled = true;
                    btn.classList.add('cursor-not-allowed');
                });
                answeredQuestions.add(questionIndex);
                updateSubmitButtonState();

            } else {
                // 오답인 경우
                selectedButton.classList.remove('bg-gray-200', 'text-gray-700');
                selectedButton.classList.add('bg-orange', 'text-white');
                selectedButton.nextElementSibling.classList.add('border-orange');
                feedbackElement.innerHTML = `<span class="text-orange font-bold">다시 한번 생각해봐요!</span>`;
            }
            feedbackElement.classList.remove('hidden');
        }
        
        // 퀴즈 제출을 처리하는 함수
        submitButton.addEventListener('click', () => {
            let score = 0;
            const questionContainers = document.querySelectorAll('#quiz-container > div');
            questionContainers.forEach((container, index) => {
                const correctAnswer = quizData[index].answer;
                const selectedButton = container.querySelector(`button[data-answer="${correctAnswer}"]`);
                // 올바른 답변을 선택하고 스타일이 적용되었는지 확인하여 점수를 계산
                if (selectedButton && selectedButton.classList.contains('bg-primary-blue')) {
                    score++;
                }
            });
            
            const percentage = (score / quizData.length) * 100;
            let message = '';
            
            if (percentage >= 90) {
                message = "👏 완벽해요! 핵심 가치를 완벽하게 이해하고 계시네요. 웅진의 훌륭한 인재가 되실 겁니다.";
            } else if (percentage >= 70) {
                message = "👍 잘했어요! 대부분의 내용을 잘 숙지하고 계십니다. 몇 가지 부분을 다시 확인하면 더욱 좋습니다.";
            } else {
                message = "🌱 조금 더 노력해봐요! 이번 교육 내용을 다시 한번 살펴보는 것을 추천합니다. 함께 성장해 나가요!";
            }
            
            document.getElementById('modal-score').innerText = `${score} / ${quizData.length} (${percentage.toFixed(0)}%)`;
            document.getElementById('modal-message').innerText = message;
            modal.style.display = 'flex';
        });

        // "닫기" 버튼 클릭 시 모달 닫기
        closeModalButton.addEventListener('click', () => {
            modal.style.display = 'none';
        });

        // "다시 풀기" 버튼 클릭 시 퀴즈 리셋
        retryQuizButton.addEventListener('click', () => {
            modal.style.display = 'none';
            renderQuiz();
            document.getElementById('quiz').scrollIntoView({ behavior: 'smooth' });
        });

        // "나가기" 버튼 클릭 시 모달 닫고 최종 페이지로 이동
        exitQuizButton.addEventListener('click', () => {
            modal.style.display = 'none';
            finalPage.classList.remove('hidden');
            finalPage.scrollIntoView({ behavior: 'smooth' });
        });

        // 페이지 로드 시 초기 퀴즈 렌더링
        renderQuiz();
        
    </script>
